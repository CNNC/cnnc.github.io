Exchange

查看某时段内的邮件：
Get-MessageTrackingLog -ResultSize Unlimited -Start "3/6/2015 8:40AM" -End "3/6/2015 1:50PM"

查看某人某个时间段邮件：
Get-MessageTrackingLog -ResultSize Unlimited -Start "3/6/2015 8:40AM" -End "3/6/2015 1:50PM" –Sender majy@corp.com


查看发送失败的邮件：
Get-MessageTrackingLog -ResultSize Unlimited -Start "3/6/2015 8:40AM" -End "3/6/2015 2:50PM" -EventId "Fail" -Sender wenyy@corp.com

数据库对所有库游标执行脚本：
DECLARE @sysname sysname; --数据库名称变量


DECLARE My_Cursor CURSOR --定义游标
FOR (select [name] from [master].[dbo].[sysdatabases]) --查出所有的数据库名称的集合放到游标中
OPEN My_Cursor; --打开游标
FETCH NEXT FROM My_Cursor INTO @sysname; --读取下一行/第一行数据并存放在变量值中
WHILE @@FETCH_STATUS = 0 --是否可继续执行循环状态
    BEGIN
        --PRINT @sysname;
        DECLARE @sql nvarchar(max); --sql命令字符串
        SET @sql='USE ['+@sysname+'];';
        --Begin
        
    --   if (CHARINDEX('NE', @sysname)>0) --我只需要修改名字包含_QtxApp的数据库
       begin
            SET @sql+='select * from dbo.Userlog'; --自定义的sql命令
            PRINT @sql;
       end
       -- else
     --     begin
      --      PRINT @sql;
     --     end
        
        --End
        EXEC(@sql); --执行字符串sql
        
        FETCH NEXT FROM My_Cursor INTO @sysname; --读取下一行数据并存放在变量值中
    END
CLOSE My_Cursor; --关闭游标
DEALLOCATE My_Cursor; --释放游标
GO





数据库对名称中含有相同内容的库游标执行脚本：

DECLARE My_Cursor CURSOR --定义游标
FOR (select [name] from [master].[dbo].[sysdatabases]) --查出所有的数据库名称的集合放到游标中
OPEN My_Cursor; --打开游标

WHILE @@FETCH_STATUS = 0 --是否可继续执行循环状态
    BEGIN
        --PRINT @sysname;
        DECLARE @sql nvarchar(max); --sql命令字符串
        SET @sql='USE ['+@sysname+'];';
        --Begin
        
       if (CHARINDEX('NE', @sysname)>0) --我只需要修改名字包含_QtxApp的数据库
        begin
            SET @sql+='select * from dbo.Userlog'; --自定义的sql命令
            PRINT @sql;
        end
        else
        begin
            PRINT @sql;
        end
        
        --End
        EXEC(@sql); --执行字符串sql
        
        FETCH NEXT FROM My_Cursor INTO @sysname; --读取下一行数据并存放在变量值中
    END
CLOSE My_Cursor; --关闭游标
DEALLOCATE My_Cursor; --释放游标
